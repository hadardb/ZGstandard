# js 风格规范

# 1. 禁止多次声明同一变量

错误示范:

var a = 3;
var a = 10;

正确示范:

var a = 3;
a = 10;

# 2. 使用骆驼拼写法命名约定

错误示范:

var my_favorite_color = "#112C85";
function do_something() {
}

正确示范:

var myFavoriteColor = "#112C85";
var _myFavoriteColor = "#112C85";

# 3. 数组内不允许出现多余空格

错误示范:

var arr = [ 'foo', 'bar' ];
var arr = ['foo', 'bar' ];

正确示范:

var arr = [];
var arr = ['foo', 'bar', 'baz'];
var arr = [['foo'], 'bar', 'baz'];

# 4. 禁止使用拖尾逗号

错误示范:

var foo = {
bar: "baz",
qux: "quux",
};

正确示范:

var foo = {
bar: "baz",
qux: "quux"
};

# 5. 禁止逗号前使用空格

错误示范:

var foo = 1 ,bar = 2;
var arr = [1 , 2];
var obj = {"foo": "bar" ,"baz": "qur"};

正确示范:

var foo = 1, bar = 2
, baz = 3;
var arr = [1, 2];

# 6. 要求逗号放在数组元素、对象属性或变量声明之后，且在同一行

错误示范:

var foo = 1
,
bar = 2;

正确示范:

var foo = 1,
bar = 2;

# 7. 禁止在计算属性中使用空格

错误示范:

obj[foo ]
obj[ 'foo']
var x = {[ b ]: a}

正确示范:

obj[foo]
obj['foo']
var x = {[b]: a}

# 8. 要求一致的 This （如果需要 this 别名，只能使用 that）

错误示范:

var that = 42;
var self = this;

正确示范:

var that = this;
var self = 42;

# 9. 禁止在函数标识符和其调用之间有空格

错误示范:

fn ();
fn
();

正确示范:

fn();

# 10. 强制行注释在代码上方，单独成行

错误示范:

1 + 1; // invalid comment
1 + 1; // invalid comment

正确示范:

// valid comment
1 + 1;
// valid comment
1 + 1;

# 11. 单行最大长度 80 个字符

# 12. 单个函数最多 100 行

# 13. 单个函数最多三个参数 更多的使用对象的方式

错误示范:

function foo (bar, baz, qux, qxx) {
    doSomething();
}

正确示范:

let a = { baz, qux, qxx}
function foo (bar, a) {
    doSomething();
}

# 14. 单行最多一个语句

错误示范:

var bar; var baz;
if (condition) { bar = 1; }
for (var i = 0; i < length; ++i) { bar = 1; }

正确示范:

var bar, baz;
if (condition) bar = 1;
for (var i = 0; i < length; ++i);

# 15. 构造函数首字母大写

错误示范:

var friend = new person();

正确示范:

var friend = new Person();

# 16. 无参的构造函数后面也必须有括号

错误示范:

var person = new Person;
var person = new (Person);

正确示范:

var person = new Person();
var person = new (Person)();

# 17. 禁止混合使用不同的操作符

错误示范:

var foo = a && b < 0 || c > 0 || d + 1 === 0;
var foo = a + b * c;

正确示范:

var foo = a || b || c;
var foo = a && b && c;
var foo = (a && b < 0) || c > 0 || d + 1 === 0;

# 18. 禁止连续赋值

错误示范:

var a = b = c = 5;
var foo = bar = "baz";

正确示范:

var a = 5;
var b = 5;
var c = 5;

# 19. 属性必须在不同的行上

错误示范:

const obj0 = { foo: "foo", bar: "bar", baz: "baz" };
const obj1 = {
    foo: "foo", bar: "bar", baz: "baz"
};

正确示范:

const obj1 = {
    foo: "foo",
    bar: "bar",
    baz: "baz"
};

# 20. 圆括号内两边没有空格

错误示范:

foo( 'bar');
var foo = ( 1 + 2 ) * 3;

正确示范:

foo('bar');
var foo = (1 + 2) * 3;

# 21. 要求中缀操作符周围有空格

错误示范:

a+b
const a={b:1};

正确示范:

a + b
const a = {b:1};

# 22. 要求在注释前有空格

错误示范:

//This is a comment with no whitespace at the beginning
/*This is a comment with no whitespace at the beginning */

正确示范:

// This is a comment with a whitespace at the beginning
/* This is a comment with a whitespace at the beginning */

# 23. 禁止重复导入模块

错误示范:

import { merge } from 'module';
import something from 'another-module';
import { find } from 'module';

正确示范:

import { merge, find } from 'module';
import something from 'another-module';

# 24. 回调函数统一使用箭头函数

错误示范:

foo(function(a) { return a; }); // ERROR

正确示范:

foo(a => a); // OK

# 25. 使用剩余参数代替 arguments

错误示范:

function foo() {
    console.log(arguments);
}

function foo(action) {
    var args = Array.prototype.slice.call(arguments, 1);
    action.apply(null, args);
}

正确示范:

function foo(...args) {
    console.log(args);
}

function foo(action, ...args) {
    action.apply(null, args); 
}

# 26. 使用模板字面量而非字符串连接

错误示范:

var str = "Hello, " + name + "!";
var str = "Time: " + (12 * 60 * 60 * 1000);

正确示范:

var str = "Hello World!";
var str = `Hello, ${name}!`;
var str = `Time: ${12 * 60 * 60 * 1000}`;

# 27. 使用单引号

错误示范:

var double = "double";
var unescaped = "a string containing 'single' quotes";

正确示范:

var single = 'single';
var backtick = `back${x}tick`; // backticks are allowed due to substitution

# 28. 句末不加分号

错误示范:

var name = "ESLint";

object.method = function() {
    // ...
};

正确示范:

var name = "ESLint"

object.method = function() {
    // ...
}

# 29. 禁止不必要的括号

错误示范:

a = (b * c);

(a * b) + c;

for (a in (b, c));

正确示范:

(0).toString();

(Object.prototype.toString.call());

({}.toString.call());

# 30. 点号跟在对象后面

错误示范:

var foo = object
.property;

正确示范:

var foo = object.
property;
var bar = object.property;

# 31. 使用点号获取参数

错误示范:

var x = foo["bar"];

正确示范:

var x = foo.bar;
var x = foo[bar];    // Property name is a variable, square-bracket notation required

# 32. 使用 === 而不是 ==

错误示范:

if (x == 42) { }

if ("" == text) { }

正确示范:

a === b
foo === true
bananas !== 1

# 33. 禁止出现空函数

错误示范:

function foo() {}

var foo = function() {};

正确示范:

function foo() {
    // do nothing.
}

var foo = function() {
    // any clear comments.
};

# 34. 禁止使用多个空格

错误示范:

var a =  1;

if(foo   === "bar") {}

正确示范:

var a = 1;

if(foo === "bar") {}

# 35. 禁止不必要的花括号嵌套

错误示范:

{}

if (foo) {
    bar();
    {
        baz();
    }
}

正确示范:

while (foo) {
    bar();
}

if (foo) {
    if (bar) {
        baz();
    }
}

# 36. 禁止自身比较

错误示范:

var x = 10;
if (x === x) {
    x = 20;
}

正确示范:

这你不懂拖出去斩咯

# 37. 禁止多余的 return 语句

错误示范:

function foo() { return; }

function foo() {
  doSomething();
  return;
}

正确示范:

function foo() { return 5; }

function foo() {
  return doSomething();
}

# 38. 禁止使用不带 await 表达式的 async 函数

错误示范:

async function foo() {
    doSomething();
}

bar(async () => {
    doSomething();
});

正确示范:

async function foo() {
    await doSomething();
}

bar(async () => {
    await doSomething();
});

# 39. 所有的 var 声明出现在它们所在的作用域顶部

错误示范:

function doSomething() {
    var first;
    if (true) {
        first = true;
    }
    var second;
}

正确示范:

function doSomething() {
    var first;
    var second; //multiple declarations are allowed at the top
    if (true) {
        first = true;
    }
}

# 40. IIFE 使用括号将函数括起来

错误示范:

var x = function () { return { y: 1 };}(); // unwrapped
var x = (function () { return { y: 1 };}()); // wrapped call expression

正确示范:

var x = (function () { return { y: 1 };})(); // wrapped function expression

# 41. var 初始化必须有值 （无法判断类型的值就赋值null）

错误示范:

function foo() {
    var bar;
    let baz;
}

正确示范:

function foo() {
    var bar = 1;
    let baz = 2;
    const qux = 3;
}

# 42. 禁止变量声明与外层作用域的变量同名

错误示范:

var a = 3;
function b() {
    var a = 10;
}

var b = function () {
    var a = 10;
}

正确示范:

if (true) {
    let a = 3;
    let b = 6;
}

let a = 5;
function b() {}

# 43. 禁止将变量初始化为 undefined

错误示范:

var foo = undefined;
let bar = undefined;

正确示范:

这你不懂拖出去斩咯

# 44. 禁止变量在禁止定义前使用（函数声明可以滞后，变量赋值必须提前，包括函数赋值。class也必须提前）

错误示范:

alert(a);
var a = 10;
new A();
class A {
}

正确示范:

f(1);
function f() {}
