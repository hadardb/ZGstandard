# js注释规范

# 函数注释

模板：

/**
 * @name:  函数名称
 * @param {type}    说明参数类型 变量名和作用
 * @return {type}   说明输出类型 变量名和作用
 */

vsCode插件推荐： koroFileHeader

# 输入输出模块

所有的输入输出函数、模块都应标注模块的作用，例如：

// 引入通用顺序表单模块
import stepsFormTpl from '../components/StepsFormTpl' 
// 第一页表单页面
import CompleteInfoFormOne from '../components/CompleteInfoFormOne' 
// 第二页表单页面
import CompleteInfoFormTwo from '../components/CompleteInfoFormTwo' 
// 第三页表单页面
import CompleteInfoFormThree from '../components/CompleteInfoFormThree' 
// 第四页表单页面
import CompleteInfoFormFour from '../components/CompleteInfoFormFour' 
import {
  // 获取用户信息接口
  getUserInfo, 
  // 提交数据接口
  allPageUpload 
} from '../api/complateFormAPI'
import mixin from '../mixin/userMixin' // 与vuex中user相关的mixin
import { mapState, mapMutations } from 'vuex'

# 单行注释

当文件较大，业务逻辑复杂或使用了难以一眼看懂的代码时应对必要的代码进行行注释，例如：

    // 1. 将迭代器+1 开始创建下一页
	let index = iterator.next()
	// 2. 将index添加到每个id中
	let newIdList = Object.values(threePageIdList).reduce((all, item) => {
		all[item] = item + index
		return all
	}, {})
	// 3. 添加index属性  模板需要这个属性对样式进行修改
	newIdList.index = index
	// 4. 模板渲染
	const threePageHTML = threePage(newIdList)
	// 5. 将模板添加到页面中
	$('#pageThreeInfo').append(threePageHTML)
	// 6. 为删除按键添加事件
	$(`#${newIdList.threeDelBtn}`).on('click', function () {
		let pageId = $(this).attr('data-pageId')
		let index = $(this).attr('data-index')
		$(`#${pageId}`).remove()
		delList[index] = index
	})

# 最终说明

注释的目的是为了他人阅读与后期更好的维护修改，但凡是有必要的就应该添加说明，包含但不限于以上几种。
