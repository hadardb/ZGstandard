# 项目结构规范

  1. vue cli3 目录结构说明
├─App.vue
├─main.js
├─views                     // 页面文件
|   ├─Home.vue
|   ├─invite.vue
|   └workbench.vue
├─store                     // vuex
|   ├─actions.js
|   ├─getters.js
|   ├─index.js
|   ├─mutations.js
|   ├─state.js
|   ├─plugin                // vuex 插件
|   |   └saveInLocal.js
|   ├─module                // vuex 模块
|   |   ├─completeInfo.js
|   |   └user.js
├─router
|   ├─index.js              // 总体引用 小项目也可将守卫直接写入
|   └router.js              // 路由列表
├─plugins                   // 插件
|    ├─axios.js
|    └element.js
├─libs                      // 通用方法
|  ├─axios.js               // axios封装
|  ├─tools.js               // 与业务无关的方法
|  └util.js                 // 与业务有关的方法
├─container                 // 意愿上想将结构与内容分离  理想上会有个结构文件夹 
|     └pageContainer.vue
├─mixin                   // mixin文件
|   └userMixin.js
├─config                    // 全局常量定义
|   └index.js
├─components                // 组件
|     ├─PageAsideMenu.vue
|     └PageHeader.vue
├─assets                    // 静态资源
|   ├─logo.png
|   ├─img
|   |  ├─bg.png
|   ├─css
|   |  ├─main.scss          // 通用scss常量
|   |  └mixin.scss          // 通用mixin
├─api                       // 接口文件夹
|  ├─app.js
|  ├─index.js
|  └inviteAPI.js
